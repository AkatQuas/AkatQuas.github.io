Lite store with Hooks

Posted on 2019-07-15 In frontend , javascript , reactjs

A brief example for using useContext, useReducer to create global store without other libraries.

/* reducer for useReducer */
const authReducer = (state, action) => {
  switch (action.type) {
    case 'increment':
      return {
        ...state,
        age: state.age + 1
      };
    case 'login':
      const { name } = action;
      return {
        ...state,
        name
      };
    default:
      return state;
  }
};

/* global context, could be use in any nested component */
const AuthContext = createContext(null);

/* Top level provider to provide the store */
const AuthProvider = ({ children }) => {
  const [state, dispatch] = useReducer(authReducer, {
    name: 'yang',
    age: 11
  });
  return (
    <AuthContext.Provider value={{ state, dispatch }}>
      {children}
    </AuthContext.Provider>
  );
};
/* consumer to use store */
const NameLabel = props => {
  const { state, dispatch } = useContext(AuthContext);

  return (
    <Fragment>
      <p>
        {state.name} @ {state.age}
      </p>
      <p>{props.label}</p>
      <button onClick={() => dispatch({ type: 'increment' })}>
        increment
      </button>
      <button
        onClick={() =>
          dispatch({
            type: 'login',
            name: Math.random().toString(16)
          })
        }
      >
        chang name
      </button>
    </Fragment>
  );
};

/* Render Entry */
function App() {
  return (
    <AuthProvider>
      <NameLabel label="example" />
    </AuthProvider>
  );
}
