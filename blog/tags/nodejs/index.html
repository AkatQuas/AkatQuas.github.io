// read-from-terminal.js
process.stdin.resume();
process.stdin.setEncoding('ascii'); // 'utf-8' also supported

let _input = '';

process.stdin.on('data', function (data) {
  _input += data;
});

process.stdin.on('end', function () {
  process.stdout.write(_input);
});
